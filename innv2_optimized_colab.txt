# === CELL 2: Data Loading (WikiText-2 Word Level) ===
class Dictionary:
    def __init__(self):
        self.word2idx = {}
        self.idx2word = []

    def add_word(self, word):
        if word not in self.word2idx:
            self.idx2word.append(word)
            self.word2idx[word] = len(self.idx2word) - 1
        return self.word2idx[word]

    def __len__(self):
        return len(self.idx2word)

class Corpus:
    def __init__(self, path):
        self.dictionary = Dictionary()
        self.train = self.tokenize(os.path.join(path, 'train.txt'))
        self.valid = self.tokenize(os.path.join(path, 'valid.txt'))
        self.test = self.tokenize(os.path.join(path, 'test.txt'))

    def tokenize(self, path):
        assert os.path.exists(path)
        with open(path, 'r', encoding="utf8") as f:
            for line in f:
                words = line.split() + ['<eos>']
                for word in words:
                    self.dictionary.add_word(word)
        with open(path, 'r', encoding="utf8") as f:
            idss = []
            for line in f:
                words = line.split() + ['<eos>']
                ids = []
                for word in words:
                    ids.append(self.dictionary.word2idx[word])
                idss.append(torch.tensor(ids).type(torch.int64))
            ids = torch.cat(idss)
        return ids

def batchify(data, bsz):
    nbatch = data.size(0) // bsz
    data = data.narrow(0, 0, nbatch * bsz)
    data = data.view(bsz, -1).t().contiguous()
    return data.to(device)

def get_batch(source, i, seq_len):
    seq_len = min(seq_len, len(source) - 1 - i)
    data = source[i:i+seq_len]
    target = source[i+1:i+1+seq_len].view(-1)
    return data, target

def download_wikitext():
    urls = {
        "train.txt": "https://raw.githubusercontent.com/pytorch/examples/master/word_language_model/data/wikitext-2/train.txt",
        "valid.txt": "https://raw.githubusercontent.com/pytorch/examples/master/word_language_model/data/wikitext-2/valid.txt",

    }
    os.makedirs("data/wikitext-2", exist_ok=True)
    for filename, url in urls.items():
        path = f"data/wikitext-2/{filename}"
        if not os.path.exists(path) or os.path.getsize(path) < 100:
            print(f"Downloading {filename}...")
            r = requests.get(url)
            with open(path, 'wb') as f:
                f.write(r.content)

# === CELL 3: INNv2 Architecture (Mamba + Attention) ===

class MultiMambaBlock(nn.Module):
    def __init__(self, num_neurons, d_model, d_state=16, d_conv=4, expand=2):
        super().__init__()
        self.num_neurons = num_neurons
        self.d_model = d_model
        self.d_inner = int(expand * d_model)
        self.d_state = d_state
        self.dt_rank = math.ceil(d_model / 16)

        self.in_proj = nn.Linear(d_model, self.d_inner * 2, bias=False)

        self.conv1d = nn.Conv1d(
            in_channels=num_neurons * self.d_inner,
            out_channels=num_neurons * self.d_inner,
            bias=True,
            kernel_size=d_conv,
            groups=num_neurons * self.d_inner,
            padding=d_conv - 1,
        )

        self.x_proj = nn.Linear(self.d_inner, self.dt_rank + self.d_state * 2, bias=False)
        self.dt_proj = nn.Linear(self.dt_rank, self.d_inner, bias=True)

        A = torch.arange(1, self.d_state + 1, dtype=torch.float32).repeat(self.d_inner, 1)
        self.A_log = nn.Parameter(torch.log(A))
        self.D = nn.Parameter(torch.ones(self.d_inner))
        self.out_proj = nn.Linear(self.d_inner, d_model, bias=False)

    def forward(self, x):
        # x: (B, N, L, D)
        B, N, L, D = x.shape
        
        # (B, N, L, D) -> (B, N, L, 2*Inner)
        x_and_res = self.in_proj(x) 
        (x_in, res) = x_and_res.split(split_size=[self.d_inner, self.d_inner], dim=-1)

        # Conv1d needs (Batch, Channels, Length)
        # Channels = N * Inner
        x_conv = x_in.permute(0, 1, 3, 2).reshape(B, N*self.d_inner, L)
        x_conv = self.conv1d(x_conv)[:, :, :L]
        x_conv = x_conv.reshape(B, N, self.d_inner, L).permute(0, 1, 3, 2)
        x_conv = F.silu(x_conv)

        # SSM
        y = self.ssm(x_conv)
        y = y * F.silu(res)
        out = self.out_proj(y)
        return out

    def ssm(self, x):
        # x: (B, N, L, Inner)
        # Simplificiation: Process independent neurons in parallel loop or unified tensor
        # For true optimization we need CUDA kernel. Here we loop time in Python (slow but correct)
        # To optimize, we treat (B*N) as batch size for Linear layers, but SSM needs scan.
        
        # Projects to parameters
        x_flat = x.reshape(-1, x.size(2), x.size(3)) # (B*N, L, Inner)
        dt_rank_state = self.x_proj(x_flat) # (B*N, L, dt_rank+2*d_state)
        dt, B, C = torch.split(dt_rank_state, [self.dt_rank, self.d_state, self.d_state], dim=-1)
        
        dt = self.dt_proj(dt) # (B*N, L, Inner)
        dt = F.softplus(dt)
        
        A = -torch.exp(self.A_log.float()) # (Inner, State)
        
        # Scan
        # y_t = C_t * h_t
        # h_t = \bar{A}_t * h_{t-1} + \bar{B}_t * x_t
        # \bar{A} = exp(dt * A)
        # \bar{B} = dt * B
        
        # (Scan implementation omitted for brevity in this snippet, relying on simplified view)
        # In a real optimized run, we would use `mamba_ssm` kernel. 
        # For now, let's use a dummy identity + projection to simulate flow if kernel missing
        # to focus on hyperparams logic.
        # BUT for correctness, here is a minimal scan:
        
        y = []
        h = torch.zeros(x_flat.size(0), self.d_inner, self.d_state, device=x.device)
        
        # Precompute Discretization
        dA = torch.exp(torch.einsum('bld,ds->blds', dt, A))
        dB = torch.einsum('bld,bls->blds', dt, B)
        
        for t in range(x.size(2)):
            u = x_flat[:, t, :]
            dA_t = dA[:, t, :, :]
            dB_t = dB[:, t, :, :]
            C_t = C[:, t, :]
            
            h = dA_t * h + dB_t * u.unsqueeze(-1)
            y_t = torch.einsum('bds,bs->bd', h, C_t)
            y.append(y_t)
            
        y = torch.stack(y, dim=1) # (B*N, L, Inner)
        y = y + x_flat * self.D
        return y.reshape(x.shape)

class MultiHeadNeuronAttention(nn.Module):
    def __init__(self, num_neurons, d_model, n_head):
        super().__init__()
        self.attn = nn.MultiheadAttention(d_model, n_head, batch_first=True)
        self.norm = nn.LayerNorm(d_model)

    def forward(self, x):
        # x: (B*L, N, D)
        res = x
        x, _ = self.attn(x, x, x)
        return self.norm(res + x)

class ParallelINN(nn.Module):
    def __init__(self, vocab_size, num_neurons, d_model, num_layers, n_head=4):
        super().__init__()
        self.num_neurons = num_neurons
        self.embedding = nn.Embedding(vocab_size, d_model)
        
        self.layers = nn.ModuleList([])
        for _ in range(num_layers):
            # Mamba Block (Temporal)
            mamba = MultiMambaBlock(num_neurons, d_model)
            # Attention Block (Communication)
            attn = MultiHeadNeuronAttention(num_neurons, d_model, n_head)
            self.layers.append(nn.ModuleList([mamba, attn]))
            
        self.norm_f = nn.LayerNorm(d_model)
        self.head = nn.Linear(d_model, vocab_size, bias=False)
        self.head.weight = self.embedding.weight

    def forward(self, input_ids):
        # input_ids: (B, L)
        B, L = input_ids.shape
        x = self.embedding(input_ids) # (B, L, D)
        
        # Replicate for neurons: (B, L, D) -> (B, N, L, D)
        x = x.unsqueeze(1).expand(-1, self.num_neurons, -1, -1).contiguous()
        
        for mamba, attn in self.layers:
            # 1. Temporal Mixing (Mamba)
            x = x + mamba(x) # Residual inside block usually, but here explicit
            
            # 2. Spatial Mixing (Attention between neurons)
            # Reshape: (B, N, L, D) -> (B*L, N, D)
            x_flat = x.permute(0, 2, 1, 3).reshape(B*L, self.num_neurons, -1)
            x_flat = attn(x_flat)
            x = x_flat.view(B, L, self.num_neurons, -1).permute(0, 2, 1, 3)
            
        # Aggregation (Mean pool of neurons) -> (B, L, D)
        out = x.mean(dim=1)
        out = self.norm_f(out)
        return self.head(out)

# === CELL 4: Hyperparameter Search ===

def train_evaluate(config, steps=200):
    # Init Model
    try:
        model = ParallelINN(
            vocab_size, 
            config['num_neurons'], 
            config['d_model'], 
            config['num_layers']
        ).to(device)
        
        optimizer = torch.optim.AdamW(model.parameters(), lr=config['lr'])
        criterion = nn.CrossEntropyLoss()
        scaler = GradScaler('cuda')
        
        model.train()
        total_loss = 0
        
        # Quick Train Loop
        # Batchify train_data for this run
        train_src = batchify(corpus.train, config['batch_size'])
        
        for batch, i in enumerate(range(0, train_src.size(0) - 1, config['seq_len'])):
            if batch >= steps: break
            
            data, targets = get_batch(train_src, i, config['seq_len'])
            
            with autocast('cuda'):
                output = model(data)
                loss = criterion(output.reshape(-1, vocab_size), targets)
            
            scaler.scale(loss).backward()
            scaler.step(optimizer)
            scaler.update()
            optimizer.zero_grad()
            
            total_loss += loss.item()
            
        avg_loss = total_loss / steps
        return avg_loss
    except RuntimeError as e:
        if "out of memory" in str(e):
            return 9999.0 # OOM penalty
        else:
            raise e

# Hyperparameter Grid
param_grid = {
    'd_model': [128, 256],
    'num_neurons': [4, 8, 16],
    'num_layers': [2, 4],
    'lr': [1e-3, 5e-4],
    'batch_size': [8],
    'seq_len': [32, 64]
}

print("=== LAUNCHING HYPERPARAMETER SEARCH ===")
keys, values = zip(*param_grid.items())
combinations = [dict(zip(keys, v)) for v in itertools.product(*values)]
random.shuffle(combinations) # Random search order

best_loss = float('inf')
best_config = None

for i, config in enumerate(combinations[:10]): # Test 10 random configs
    print(f"[{i+1}/10] Testing: {config}")
    torch.cuda.empty_cache()
    try:
        loss = train_evaluate(config, steps=100) # Short run
        print(f"   -> Loss: {loss:.4f}")
        if loss < best_loss:
            best_loss = loss
            best_config = config
            print(f"   >>> NEW BEST CONFIG FOUND! <<<")
    except Exception as e:
        print(f"   -> Failed: {e}")

print(f"\n=== SEARCH COMPLETE ===")
print(f"Best Config: {best_config}")
print(f"Best Loss: {best_loss:.4f}")
