# @title ⚔️ THE ARENA V2: INN vs Transformer vs Mamba (Stabilized)
# @markdown **Objectif :** Reproduire la performance de l'INN V5 (Word Level) mais sans les oscillations finales.
# @markdown - **Modif 1 :** Accumulation = 16 (Batch 128) -> Stabilité.
# @markdown - **Modif 2 :** Epochs = 3 -> Temps de convergence.
# @markdown - **Modif 3 :** LR = 3e-4 + Hard Decay.

# --- 2. CONFIGURATION (STABILITY PATCH) ---
CONFIG = {
    'dataset': 'wikitext-103-v1',

# --- 3. DATA LOADING ---
class ArenaDataset(Dataset):
    def __init__(self, split, vocab=None, limit=None):
        print(f"📚 Loading WikiText-103 ({split})...")
        dataset = load_dataset("wikitext", "wikitext-103-v1", split=split)
        text = "\n".join(dataset['text'])
        tokens = text.replace('\n', ' <eos> ').split()
        tokens = [t for t in tokens if t.strip()]
        
        if vocab is None:
            counts = Counter(tokens)
            most_common = counts.most_common(limit)
            self.vocab = sorted([w for w, c in most_common])
            self.word2idx = {w: i for i, w in enumerate(self.vocab)}
            self.unk_idx = len(self.vocab)
            self.vocab_size = len(self.vocab) + 1
        else:
            self.vocab = vocab
            self.word2idx = {w: i for i, w in enumerate(self.vocab)}
            self.vocab_size = len(self.vocab) + 1
            self.unk_idx = self.vocab_size - 1

        self.data = torch.tensor([self.word2idx.get(t, self.unk_idx) for t in tokens], dtype=torch.long)
        self.seq_len = CONFIG['seq_len']
        print(f"✅ {split.upper()} Ready. {len(self.data)} tokens.")

    def __len__(self): return len(self.data) // self.seq_len
    def __getitem__(self, idx):
        start = idx * self.seq_len
        chunk = self.data[start:start+self.seq_len+1]
        if len(chunk) < self.seq_len+1:
            chunk = torch.cat([chunk, torch.zeros(self.seq_len+1-len(chunk), dtype=torch.long)])
        return chunk[:-1], chunk[1:]

# --- 4. THE GLADIATORS ---

# 🔴 INN V5 (Champion)
class INN_V5(nn.Module):
    def __init__(self, vocab_size, d_model=512, n_neurons=64, n_layers=4):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.neuron_id = nn.Parameter(torch.randn(1, 1, n_neurons, d_model))
        self.layers = nn.ModuleList([
            nn.ModuleDict({
                'norm_in': nn.LayerNorm(d_model),
                'mamba': Mamba(d_model=d_model, d_state=16, d_conv=4, expand=2),
                'comm_norm': nn.LayerNorm(d_model),
                'comm': nn.MultiheadAttention(d_model, num_heads=8, batch_first=True)
            }) for _ in range(n_layers)
        ])
        self.readout_attn = nn.MultiheadAttention(d_model, num_heads=4, batch_first=True)
        self.readout_query = nn.Parameter(torch.randn(1, 1, d_model))
        self.norm_f = nn.LayerNorm(d_model)
        self.head = nn.Linear(d_model, vocab_size)
        self.n_neurons = n_neurons

    def forward(self, x):
        B, L = x.shape
        h = self.embedding(x).unsqueeze(2).expand(-1, -1, self.n_neurons, -1)
        h = h + self.neuron_id
        
        for layer in self.layers:
            B, L, N, D = h.shape
            h_flat = h.view(B*N, L, D)
            h_mamba = layer['mamba'](layer['norm_in'](h_flat)).view(B, L, N, D)
            h = h + h_mamba
            
            h_flat = h.view(B*L, N, D)
            h_norm = layer['comm_norm'](h_flat)
            attn, _ = layer['comm'](h_norm, h_norm, h_norm)
            h = h + attn.view(B, L, N, D)
            
        B, L, N, D = h.shape
        flat_h = h.view(B*L, N, D)
        q = self.readout_query.expand(B*L, -1, -1)
        out, _ = self.readout_attn(q, flat_h, flat_h)
        out = self.norm_f(out.view(B, L, D))
        return self.head(out)

# 🔵 Transformer Baseline
class Transformer_Baseline(nn.Module):
    def __init__(self, vocab_size, d_model=512, n_layers=6):
        super().__init__()
        self.token_embedding = nn.Embedding(vocab_size, d_model)
        self.pos_embedding = nn.Parameter(torch.zeros(1, 1024, d_model))
        encoder_layer = nn.TransformerEncoderLayer(d_model=d_model, nhead=8, dim_feedforward=d_model*4, batch_first=True, norm_first=True)
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers=n_layers)
        self.head = nn.Linear(d_model, vocab_size)

    def forward(self, x):
        x = self.token_embedding(x) + self.pos_embedding[:, :x.size(1), :]
        mask = nn.Transformer.generate_square_subsequent_mask(x.size(1)).to(x.device)
        x = self.transformer(x, mask=mask, is_causal=True)
        return self.head(x)

# 🟢 Mamba Baseline
class Mamba_Baseline(nn.Module):
    def __init__(self, vocab_size, d_model=512, n_layers=6):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.layers = nn.ModuleList([
            Mamba(d_model=d_model, d_state=16, d_conv=4, expand=2)
            for _ in range(n_layers)
        ])
        self.norm_f = nn.LayerNorm(d_model)
        self.head = nn.Linear(d_model, vocab_size)

    def forward(self, x):
        x = self.embedding(x)
        for layer in self.layers:
            x = x + layer(x)
        x = self.norm_f(x)
        return self.head(x)

# --- 5. BATTLE LOOP ---
def train_model(name, model, train_dl, valid_dl, vocab_size):
    print(f"\n🥊 FIGHTER: {name}")
    print(f"   Params: {sum(p.numel() for p in model.parameters())/1e6:.2f}M")
    
    model = model.to(device)
    optimizer = optim.AdamW(model.parameters(), lr=CONFIG['lr'], weight_decay=0.01)
    
    # SCHEDULER STABILISÉ
    total_steps = len(train_dl) * CONFIG['epochs'] // CONFIG['accum_steps']
    scheduler = torch.optim.lr_scheduler.OneCycleLR(
        optimizer, 
        max_lr=CONFIG['lr'], 
        total_steps=total_steps,
        pct_start=0.1, # Warmup 10%
        div_factor=25,
        final_div_factor=10000 # <--- FORCE LE LR À ZÉRO À LA FIN
    )
    
    scaler = GradScaler()
    criterion = nn.CrossEntropyLoss()
    
    model.train()
    start_time = time.time()
    
    for epoch in range(CONFIG['epochs']):
        for i, (x, y) in enumerate(train_dl):
            x, y = x.to(device), y.to(device)
            
            with autocast():
                output = model(x)
                loss = criterion(output.view(-1, vocab_size), y.view(-1))
                loss = loss / CONFIG['accum_steps']
            
            scaler.scale(loss).backward()
            
            if (i + 1) % CONFIG['accum_steps'] == 0:
                scaler.unscale_(optimizer)
                torch.nn.utils.clip_grad_norm_(model.parameters(), CONFIG['grad_clip'])
                scaler.step(optimizer)
                scaler.update()
                scheduler.step()
                optimizer.zero_grad()
                
                step = (i + 1) // CONFIG['accum_steps']
                # Log plus fréquent pour surveiller
                if step % 50 == 0:
                    ppl = math.exp(loss.item() * CONFIG['accum_steps'])
                    elapsed = time.time() - start_time
                    print(f"   Ep {epoch+1} | Step {step} | PPL: {ppl:.2f} | Time: {elapsed:.0f}s")
        
        # Validation Rapide à chaque Epoch
        print(f"   🏁 Validating Epoch {epoch+1}...")
        model.eval()
        val_loss = 0
        steps = 0
        with torch.no_grad():
            for x, y in valid_dl:
                x, y = x.to(device), y.to(device)
                with autocast():
                    out = model(x)
                    val_loss += criterion(out.view(-1, vocab_size), y.view(-1)).item()
                steps += 1
                if steps > 200: break
        
        valid_ppl = math.exp(val_loss / steps)
        print(f"   🏆 END EPOCH {epoch+1} | VALID PPL: {valid_ppl:.2f} 🏆")
        model.train()
    
    return valid_ppl
