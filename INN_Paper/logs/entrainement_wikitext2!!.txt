CONFIG = {
    'dataset': 'wikitext-2',
    'vocab_size': 33278,
    'd_model': 512,        # GROS CERVEAU
    'n_neurons': 64,       # 64 Experts

# --- 2. DATA ---
def get_wikitext2():
    dataset = load_dataset("wikitext", "wikitext-2-v1")
    train_text = "\n".join(dataset['train']['text'])
    valid_text = "\n".join(dataset['validation']['text'])
    return train_text, valid_text

class WordLevelDataset(Dataset):
    def __init__(self, text, seq_len, vocab=None):
        tokens = text.replace('\n', ' <eos> ').split()
        tokens = [t for t in tokens if t.strip()]
        if vocab is None:
            counts = Counter(tokens)
            self.vocab = sorted(list(counts.keys()))
            self.word2idx = {w: i for i, w in enumerate(self.vocab)}
        else:
            self.vocab = vocab
            self.word2idx = {w: i for i, w in enumerate(self.vocab)}
        self.data = torch.tensor([self.word2idx.get(t, 0) for t in tokens], dtype=torch.long)
        self.seq_len = seq_len

    def __len__(self): return len(self.data) // self.seq_len
    def __getitem__(self, idx):
        start = idx * self.seq_len
        chunk = self.data[start:start+self.seq_len+1]
        if len(chunk) < self.seq_len+1:
            chunk = torch.cat([chunk, torch.zeros(self.seq_len+1-len(chunk), dtype=torch.long)])
        return chunk[:-1], chunk[1:]

# --- 3. MODEL: STABILIZED INN ---

class StabilizedCommunication(nn.Module):
    def __init__(self, d_model, n_heads, dropout=0.1):
        super().__init__()
        self.norm_pre = nn.LayerNorm(d_model) # Sandwich Norm 1
        self.attn = nn.MultiheadAttention(d_model, n_heads, dropout=dropout, batch_first=True)
        self.norm_post = nn.LayerNorm(d_model) # Sandwich Norm 2
        self.dropout = nn.Dropout(dropout)

    def forward(self, x):
        B, L, N, D = x.shape
        x_flat = x.view(B * L, N, D)
        
        # Pre-Norm
        x_norm = self.norm_pre(x_flat)
        
        # Attention
        attn_out, _ = self.attn(x_norm, x_norm, x_norm)
        
        # Residual + Post-Norm
        x_out = self.norm_post(x_flat + self.dropout(attn_out))
        
        return x_out.view(B, L, N, D)

class WeightedReadout(nn.Module):
    """ Smart Pooling: Chaque neurone prédit sa propre confiance """
    def __init__(self, d_model):
        super().__init__()
        self.confidence_proj = nn.Linear(d_model, 1)
        
    def forward(self, x):
        # x: (B, L, N, D)
        
        # Calcul de la confiance (B, L, N, 1)
        confidence_scores = self.confidence_proj(x)
        weights = F.softmax(confidence_scores, dim=2) # Somme à 1 sur l'axe des neurones
        
        # Somme pondérée : Somme(Poids * Neurone)
        weighted_out = (x * weights).sum(dim=2) # (B, L, D)
        return weighted_out

class INN_Ultimate(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.embedding = nn.Embedding(config['vocab_size'], config['d_model'])
        
        # Neuron Identity
        self.neuron_id = nn.Parameter(torch.randn(1, 1, config['n_neurons'], config['d_model']))
        
        self.layers = nn.ModuleList([])
        for _ in range(config['n_layers']):
            self.layers.append(nn.ModuleDict({
                'norm_in': nn.LayerNorm(config['d_model']), # Pre-Norm Mamba
                'mamba': Mamba(d_model=config['d_model'], d_state=16, d_conv=4, expand=2),
                'comm': StabilizedCommunication(config['d_model'], config['n_heads'], config['dropout'])
            }))
        
        self.readout = WeightedReadout(config['d_model'])
        self.norm_final = nn.LayerNorm(config['d_model'])
        self.output_head = nn.Linear(config['d_model'], config['vocab_size'])

    def forward(self, x):
        B, L = x.shape
        N = self.config['n_neurons']
        
        h = self.embedding(x)
        h = h.unsqueeze(2).expand(-1, -1, N, -1)
        h = h + self.neuron_id
        
        for layer in self.layers:
            # Mamba Block (with Pre-Norm)
            h_flat = h.view(B * N, L, -1)
            h_norm = layer['norm_in'](h_flat)
            h_mamba = layer['mamba'](h_norm)
            h = h + h_mamba.view(B, L, N, -1) # Residual manually managed here if needed, but Mamba usually handles internals. Let's do clean residual.
            
            # Communication Block (Self-contained residual)
            h = layer['comm'](h)
            
        h_final = self.readout(h)
        h_final = self.norm_final(h_final)
        
        return self.output_head(h_final)

# --- 4. RUN ---
def run():
    train_txt, valid_txt = get_wikitext2()
    train_ds = WordLevelDataset(train_txt, CONFIG['seq_len'])
    valid_ds = WordLevelDataset(valid_txt, CONFIG['seq_len'], vocab=train_ds.vocab)
    train_dl = DataLoader(train_ds, batch_size=CONFIG['batch_size'], shuffle=True)
    valid_dl = DataLoader(valid_ds, batch_size=CONFIG['batch_size'])
    
    CONFIG['vocab_size'] = len(train_ds.vocab)
    model = INN_Ultimate(CONFIG).to(device)
    criterion = nn.CrossEntropyLoss()
    
    # Optimizer & Scheduler setup
    optimizer = optim.AdamW(model.parameters(), lr=CONFIG['max_lr'], weight_decay=0.01)
    
    total_steps = len(train_dl) * CONFIG['epochs'] // CONFIG['accum_steps']
    scheduler = torch.optim.lr_scheduler.OneCycleLR(
        optimizer, 
        max_lr=CONFIG['max_lr'], 
        total_steps=total_steps,
        pct_start=0.1, # 10% Warmup
        anneal_strategy='cos'
    )
    
    print(f"🚀 INN V5 (ULTIMATE) | Params: {sum(p.numel() for p in model.parameters())/1e6:.2f}M")
    print(f"📅 Training for {CONFIG['epochs']} epochs with OneCycleLR.")
    
    model.train()
    start_time = time.time()
    
    optimizer.zero_grad()
    
    for epoch in range(CONFIG['epochs']):
        epoch_loss = 0
        
        for i, (x, y) in enumerate(train_dl):
            x, y = x.to(device), y.to(device)
            
            # Forward
            output = model(x)
            loss = criterion(output.view(-1, CONFIG['vocab_size']), y.view(-1))
            
            # Normalize loss for accumulation
            loss = loss / CONFIG['accum_steps']
            loss.backward()
            
            epoch_loss += loss.item() * CONFIG['accum_steps']
            
            # Step only every N batches
            if (i + 1) % CONFIG['accum_steps'] == 0:
                torch.nn.utils.clip_grad_norm_(model.parameters(), CONFIG['grad_clip'])
                optimizer.step()
                scheduler.step()
                optimizer.zero_grad()
                
                # Logging (on actual steps)
                actual_step = (i + 1) // CONFIG['accum_steps']
                if actual_step % 20 == 0:
                    elapsed = max(time.time() - start_time, 1e-5)
                    current_lr = scheduler.get_last_lr()[0]
                    # Estimate raw PPL from current batch loss (approx)
                    ppl = math.exp(loss.item() * CONFIG['accum_steps'])
                    print(f"Ep {epoch+1} | Step {actual_step} | LR: {current_lr:.2e} | PPL: {ppl:.2f}")

        # Validation Full
        model.eval()
        val_loss = 0
        with torch.no_grad():
            for x, y in valid_dl:
                x, y = x.to(device), y.to(device)
                val_loss += criterion(model(x).view(-1, CONFIG['vocab_size']), y.view(-1)).item()
        valid_ppl = math.exp(val_loss/len(valid_dl))
        print(f"🏆 END EPOCH {epoch+1} | VALID PPL: {valid_ppl:.2f} 🏆")
        model.train()

if __name__ == "__main__":
    run()
